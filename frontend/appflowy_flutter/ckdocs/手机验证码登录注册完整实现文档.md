# 手机验证码登录注册完整实现文档

## 概述
本文档详细记录了在 AppFlowy 项目中实现手机验证码登录注册功能的完整流程，该功能完全兼容现有的邮箱验证码系统，用户可以使用手机号或邮箱进行注册和登录。

实现日期：2025年10月6日
项目环境：开发环境 (docker-compose-dev.yml)

---

## 一、整体架构

### 1.1 系统组件
- **前端**: Flutter (AppFlowy客户端)
- **认证服务**: GoTrue (PonyNotes-gotrue - 自定义版本)
- **后端服务**: AppFlowy-Cloud (Rust)
- **数据库**: 
  - PostgreSQL (云端数据库)
  - SQLite (本地客户端数据库)

### 1.2 实现原理
1. 客户端输入手机号或邮箱
2. 客户端自动识别输入类型（邮箱/手机号）
3. 发送验证码请求到 GoTrue
4. GoTrue 识别并发送短信或邮件验证码
5. 用户输入验证码进行验证
6. GoTrue 验证成功后返回 token
7. AppFlowy-Cloud 根据 token 创建或登录用户
8. 支持邮箱为空，仅使用手机号创建账户

```go
func (p *VerifyParams) Validate(r *http.Request, a *API) error {
	// ...
	case http.MethodPost:
		if (p.Token == "" && p.TokenHash == "") || (p.Token != "" && p.TokenHash != "") {
			return apierrors.NewBadRequestError(
				apierrors.ErrorCodeValidationFailed, 
				"Verify requires either a token or a token hash"
			)
		}
		if p.Token != "" {
			// 自动检测：email 字段包含手机号
			if p.Email != "" && isPhoneNumber(p.Email) {
				p.Phone = p.Email
				p.Email = ""
			}
			
			// 根据类型生成 token hash
			if isPhoneOtpVerification(p) {
				p.Phone, err = validatePhone(p.Phone)
				if err != nil {
					return err
				}
				p.TokenHash = crypto.GenerateTokenHash(p.Phone, p.Token)
			} else if isEmailOtpVerification(p) {
				p.Email, err = a.validateEmail(p.Email)
				if err != nil {
					return apierrors.NewUnprocessableEntityError(
						apierrors.ErrorCodeValidationFailed, 
						"Invalid email format"
					).WithInternalError(err)
				}
				p.TokenHash = crypto.GenerateTokenHash(p.Email, p.Token)
			} else {
				return apierrors.NewBadRequestError(
					apierrors.ErrorCodeValidationFailed, 
					"Only an email address or phone number should be provided on verify"
				)
			}
		}
	// ...
}
```

```go
func isOtpValid(actual, expected string, sentAt *time.Time, otpExp uint) bool {
	// 🎯 开发环境：跳过验证码验证
```

### 3.3 辅助函数

```go
// isPhoneNumber 检查字符串是否为手机号
// 用于自动检测 email 字段实际包含手机号的情况
func isPhoneNumber(s string) bool {
	// 简单启发式规则：
	// - 长度至少 10 位
	// - 不包含 @ 符号
	// - 包含足够多的数字
	if len(s) >= 10 {
		hasAt := false
		digitCount := 0
		for _, c := range s {
			if c == '@' {
				hasAt = true
				break
			}
			if c >= '0' && c <= '9' {
				digitCount++
			}
		}
		// 不包含 @ 且至少有 10 个数字
		return !hasAt && digitCount >= 10
	}
	return false
}
```

```dart
class _EmailLoginSectionState extends State<_EmailLoginSection> {
  final TextEditingController _controller = TextEditingController();
  final FocusNode _focusNode = FocusNode();
  bool _agreedToTerms = true;

  // 验证手机号格式（中国手机号）
  bool _isValidPhone(String phone) {
    final phoneRegex = RegExp(r'^1[3-9]\d{9}$');
    return phoneRegex.hasMatch(phone);
  }

  // 验证邮箱格式
  bool _isValidEmail(String email) {
    final emailRegex = RegExp(r'^[\w-\.]+@([\w-]+\.)+[\w-]{2,4}$');
    return emailRegex.hasMatch(email);
  }

  // 验证邮箱或手机号
  bool _isValidEmailOrPhone(String input) {
    return _isValidEmail(input) || _isValidPhone(input);
  }

  // 清理手机号格式，移除国际区号
  String _cleanPhoneNumber(String phone) {
    String cleanPhone = phone.trim();
    if (cleanPhone.startsWith('+86')) {
      cleanPhone = cleanPhone.substring(3);
    } else if (cleanPhone.startsWith('86') && cleanPhone.length == 13) {
      cleanPhone = cleanPhone.substring(2);
    }
    return cleanPhone;
  }
```

### 附录B：数据库表结构
```sql
-- af_user 表结构（关键字段）
CREATE TABLE af_user (
    uid BIGINT PRIMARY KEY,
    uuid UUID NOT NULL UNIQUE,
    email TEXT,                    -- 可空
    phone TEXT,                    -- 可空（新增）
    name TEXT NOT NULL,
    created_at TIMESTAMP NOT NULL DEFAULT NOW(),
    updated_at TIMESTAMP NOT NULL DEFAULT NOW(),
    
    -- 约束：至少有一种联系方式
    CONSTRAINT af_user_email_or_phone_check 
    CHECK (
        (email IS NOT NULL AND email != '') OR 
        (phone IS NOT NULL AND phone != '')
    )
);

-- 索引
CREATE UNIQUE INDEX af_user_email_key ON af_user (email) 
WHERE email IS NOT NULL AND email != '';

CREATE UNIQUE INDEX af_user_phone_key ON af_user (phone) 
WHERE phone IS NOT NULL AND phone != '';

CREATE INDEX af_user_email_idx ON af_user (email) 
WHERE email IS NOT NULL AND email != '';

CREATE INDEX af_user_phone_idx ON af_user (phone) 
WHERE phone IS NOT NULL AND phone != '';
```
